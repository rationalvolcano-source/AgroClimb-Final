function CheckIn(){
  const [log,setLog]=useLocalVar('checkins', []);
  const today = todayKey();
  const practiced = log.includes(today);
  function toggle(){
    if(practiced){ setLog(log.filter(d=>d!==today)); }
    else { setLog([...log, today]); }
  }
  // streak
  let streak=0; const now=new Date();
  while(true){
    const d = new Date(now.getFullYear(), now.getMonth(), now.getDate() - streak);
    const key = d.toISOString().slice(0,10);
    if(log.includes(key)) streak++; else break;
  }
  return (
    <section className="card">
      <div className="tileHead">
        <div>
          <div style={{fontWeight:800,fontSize:20}}>Daily Wellness Check‑in</div>
          <div className="tip">Mark your practice once a day. Builds consistency and motivation.</div>
        </div>
        <label className="check"><input type="checkbox" checked={practiced} onChange={toggle} /> I practiced today</label>
      </div>
      <div className="kpi">
        <div className="box"><div className="h">Current Streak</div><div className="v">{streak} day{streak===1?"":"s"}</div></div>
        <div className="box"><div className="h">Total Check‑ins</div><div className="v">{log.length}</div></div>
      </div>
      <div className="tip">Data is stored locally on this device. You can safely refresh or come back later.</div>
    </section>
  );
}
